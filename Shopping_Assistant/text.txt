# ── Helpers ───────────────────────────────────────────────────────────────

def stars(rating: float) -> str:
    full = int(rating)
    half = 1 if (rating - full) >= 0.5 else 0
    empty = 5 - full - half
    return "★" * full + "✦" * half + "☆" * empty


def fmt_price(price: float) -> str:
    if price >= 100000:
        return f"₹{price/100000:.1f}L"
    elif price >= 1000:
        return f"₹{price:,.0f}"
    return f"₹{price:.0f}"


CATEGORY_ICONS = {
    "smartphone": "📱", "laptop": "💻", "headphone": "🎧",
    "tablet": "📲", "tv": "📺", "camera": "📷", "watch": "⌚",
    "earbuds": "🎵", "speaker": "🔊", "keyboard": "⌨️",
    "mouse": "🖱️", "monitor": "🖥️", "default": "🛍️",
}

def product_icon(name: str) -> str:
    name_lower = name.lower()
    for k, v in CATEGORY_ICONS.items():
        if k in name_lower:
            return v
    return CATEGORY_ICONS["default"]

        result = None

        # Simulate progressive steps while running

        with st.spinner("Running AI agent pipeline..."):

        for i, step in enumerate(steps[2:], start=2):

# ── Helpers ───────────────────────────────────────────────────────────────

def stars(rating: float) -> str:
    full = int(rating)
    half = 1 if (rating - full) >= 0.5 else 0
    empty = 5 - full - half
    return "★" * full + "✦" * half + "☆" * empty


def fmt_price(price: float) -> str:
    if price >= 100000:
        return f"₹{price/100000:.1f}L"
    elif price >= 1000:
        return f"₹{price:,.0f}"
    return f"₹{price:.0f}"


CATEGORY_ICONS = {
    "smartphone": "📱", "laptop": "💻", "headphone": "🎧",
    "tablet": "📲", "tv": "📺", "camera": "📷", "watch": "⌚",
    "earbuds": "🎵", "speaker": "🔊", "keyboard": "⌨️",
    "mouse": "🖱️", "monitor": "🖥️", "default": "🛍️",
}

def product_icon(name: str) -> str:
    name_lower = name.lower()
    for k, v in CATEGORY_ICONS.items():
        if k in name_lower:
            return v
    return CATEGORY_ICONS["default"]

# ── Run pipeline ────────────────────────────────────────────────────────────

if search_clicked and user_query:
    if not os.getenv("GROQ_API_KEY") or not os.getenv("TAVILY_API_KEY"):
        st.error("⚠️ Please add your **GROQ_API_KEY** and **TAVILY_API_KEY** in the sidebar.")
        st.stop()

    # Enrich query with sidebar filters
    full_query = user_query
    if f"₹{budget:,}" not in user_query and "under" not in user_query.lower():
        full_query += f" under ₹{budget:,}"
    if brand != "Any" and brand.lower() not in user_query.lower():
        full_query += f" {brand}"
    if category != "All" and category.lower() not in user_query.lower():
        full_query += f" {category}"

    # Loading UI
    status_container = st.empty()
    progress_bar = st.progress(0)
    steps = [
        "🔍 Understanding your query...",
        "🛒 Searching for products...",
        "⚖️ Comparing products...",
        "💰 Optimizing for budget...",
        "📝 Analysing reviews...",
        "🏆 Generating recommendation...",
        "📦 Assembling final output...",
    ]
    step_placeholder = st.empty()

    def update_progress(step_idx, msg):
        progress_bar.progress((step_idx + 1) / len(steps))
        step_placeholder.markdown(f"""
        <div class="agent-step">
            <span style="color: #6366f1; font-size: 16px;">⚡</span>
            <span>{msg}</span>
        </div>
        """, unsafe_allow_html=True)

    for i, step in enumerate(steps[:2]):
        update_progress(i, step)
        time.sleep(0.1)

    # Import and run
    try:
        from graph.shopping_graph import run_shopping_assistant
        result = None

        # Simulate progressive steps while running
        update_progress(1, "🛒 Searching for products...")

        with st.spinner("Running AI agent pipeline..."):
            result = run_shopping_assistant(full_query)

        for i, step in enumerate(steps[2:], start=2):
            update_progress(i, step)
            time.sleep(0.05)

        progress_bar.progress(1.0)
        step_placeholder.empty()
        status_container.empty()

    except Exception as e:
        progress_bar.empty()
        step_placeholder.empty()
        st.error(f"❌ Pipeline error: {e}")
        st.info("💡 Check your API keys and ensure all dependencies are installed.")
        st.stop()

    if not result or "error" in result:
        st.error(f"❌ {result.get('error', 'Unknown error occurred')}")
        st.stop()

    # ── Store in session ──
    st.session_state["result"] = result
    st.session_state["query"] = user_query
